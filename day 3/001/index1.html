        VD thuộc tính màu săc: xanh, đỏ, tím , vàng
        thuộc tính font-size:12px,20px

        tên thuộc tính 1:giá trị 1
        tên thuộc tính 2:giá trị 2
        tên thuộc tính 3:giá trị 3
        }
